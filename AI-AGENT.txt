const { Client } = require('ssh2');

function runSSHCommand(command) {
  return new Promise((resolve, reject) => {
    const conn = new Client();

    conn.on('ready', () => {
      conn.exec(command, (err, stream) => {
        if (err) reject(err);

        let data = '';

        stream.on('data', chunk => {
          data += chunk.toString();
        });

        stream.on('close', () => {
          conn.end();
          resolve(data);
        });
      });
    });

      username,
      password
    });
  });
}
