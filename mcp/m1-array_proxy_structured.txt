#!/usr/bin/env python3
import sys

def get_command_data(filename, command_name):
    commands = {}
    current_cmd = None

    with open(filename, "r") as f:
        for line in f:
            line = line.rstrip()

            # Detect command start
            if line.startswith("+"):
                current_cmd = line[1:].strip()
                commands[current_cmd] = []
            else:
                if current_cmd:
                    commands[current_cmd].append(line)

    if command_name in commands:
        return "\n".join(commands[command_name])
    else:
        return "Command not found"


if len(sys.argv) != 3:
    print(f"Usage: {sys.argv[0]} <datafile> <command>")
    sys.exit(1)

file_name = sys.argv[1]
cmd = sys.argv[2]

result = get_command_data(file_name, cmd)
print(result)
